export interface SearchCriteria {

  tipoArtefacto: string;
  forma: string;

  // criterios de dimensiones
  // para diametro
  diametroCalibreMax: number;
  diametroCalibreMin: number;

  longitudMax: number;
  longitudMin: number;
  // para fondo
  fondoMax: number;
  fondoMin: number;
  // para altura
  alturaMax: number;
  alturaMin: number;
  //otros criterios de Artefacto
  color: string;
  fabricacion: string;
  propositoArtefacto: string;

  //de minas
  materialEnvuelta: string;
  situacionEspoleta: string;

  // De municion
  longitudEspoletadoMax: number;
  longitudEspoletadoMin: number;
  //de proyectil
  tipoCulote: string;
  // de bombas de aviacion
  pinturaProteccionTermica: boolean;
  distanciaAnillasMax: number;
  distanciaAnillasMin: number;
  tipoEstabilizador: string;
  venterolEnEstabilizador: boolean;
  // de Cohete
  colorBandaMotor: string;
  // de Misil
  sistemaGuia: string;
  // de Granada
  medioLanzamiento: string;

  // de espoleta
  sistemaIniciacion: string;
}
