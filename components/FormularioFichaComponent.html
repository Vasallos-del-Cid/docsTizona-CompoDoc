import { Component, OnInit, ViewChild } from "@angular/core";

import { ActivatedRoute } from "@angular/router";
import { Observable, of } from "rxjs";

import { tiposEnvironment } from "src/environments/tiposEnvironment";

import { FichaService } from "../services/ficha.service";

import { FormulariosTipo } from "src/app/comunes/formularios-dinamicos/models/formularios-tipo";
import { FichaPlantillaFormService } from "../services/ficha-plantilla-form.service";

import { ArchivosComponent } from "src/app/archivos/archivos/archivos.component";
import { FotografiasComponent } from "src/app/fotografias/fotografias/fotografias.component";
import { FormGroup } from "@angular/forms";
import Swal from "sweetalert2";

/**
 * Componente que genera mediante formularios dinamicos los elementos de input para la carga en el Back de los distintos tipos de fichas
 */
@Component({
  selector: "app-formulario-ficha",
  templateUrl: "./formulario-ficha.component.html",
  styles: [],
})
export class FormularioFichaComponent implements OnInit {

  id: number = this.activatedRoute.snapshot.params["id"];
  //ViewChild para poder acceder a los metodos del hijo
  @ViewChild(FotografiasComponent, { read: false, static: true })
  hijo: FotografiasComponent;
  @ViewChild(ArchivosComponent, { read: false, static: true })
  hijoArchivo: ArchivosComponent;

  elementosArtefacto$: Observable<FormulariosTipo<any>[]>;

  formulario: FormGroup;

  // plantillas de formas
  elementosFormaEsfera$: Observable<FormulariosTipo<any>[]>;
  elementosFormaCilindro$: Observable<FormulariosTipo<any>[]>;
  elementosFormaCubo$: Observable<FormulariosTipo<any>[]>;

  // plaltillas de tipos
  elementosFormaEspoleta$: Observable<FormulariosTipo<any>[]>;
  elementosFormaMina$: Observable<FormulariosTipo<any>[]>;
  elementosFormaMunicion$: Observable<FormulariosTipo<any>[]>;
  elementosFormaProyectil$: Observable<FormulariosTipo<any>[]>;
  elementosFormaCohete$: Observable<FormulariosTipo<any>[]>;
  elementosFormaMisil$: Observable<FormulariosTipo<any>[]>;
  elementosFormaBomba$: Observable<FormulariosTipo<any>[]>;
  elementosFormaGranada$: Observable<FormulariosTipo<any>[]>;

  // selectores para visualizacion
  flagForma = {
    cilindro: false,
    esfera: false,
    cubo: false,
  };

  flagTipo = {
    espoleta: false,
    mina: false,
    municion: false,
    proyectil: false,
    cohete: false,
    misil: false,
    bomba: false,
    granada: false,
  };

  artefacto: any;

  constructor(
    private fichaService: FichaService,
    private plantillaFicha: FichaPlantillaFormService,
    private activatedRoute: ActivatedRoute
  ) {
    this.elementosArtefacto$ = of(this.plantillaFicha.getArrayArtefacto());
    this.elementosFormaCilindro$ = of(
      this.plantillaFicha.getArrayForma("cilindro")
    );
    this.elementosFormaEsfera$ = of(
      this.plantillaFicha.getArrayForma("esfera")
    );
    this.elementosFormaCubo$ = of(this.plantillaFicha.getArrayForma(""));
    this.elementosFormaEspoleta$ = of(this.plantillaFicha.getArrayEspoleta());
    this.elementosFormaMina$ = of(this.plantillaFicha.getArrayMina());
    this.elementosFormaMunicion$ = of(this.plantillaFicha.getArrayMunicion());
    this.elementosFormaProyectil$ = of(this.plantillaFicha.getArrayProyectil());
    this.elementosFormaCohete$ = of(this.plantillaFicha.getArrayCohete());
    this.elementosFormaMisil$ = of(this.plantillaFicha.getArrayMisil());
    this.elementosFormaBomba$ = of(this.plantillaFicha.getArrayBomba());
    this.elementosFormaGranada$ = of(this.plantillaFicha.getArrayGranada());
  }

  /**
   * metodo inicial que muestra un disclaimer
   */
  ngOnInit() {
    Swal.fire({
      title: "¡Información Importante!",
      text: "Recuerde que en esta aplicación NO ESTA PERMITIDO cargar información clasificada o susceptible de serlo. Para más información consulte la pestaña 'Normas de uso'.",
      icon: "warning",
    });
  }

  /**
   * metodo que recupera los propositos de cada tipo de artefacto en funcion de la seleccion del input tipo de artefacto
   */
  seleccionTipo() {
    this.flagTipo.espoleta = false;
    this.flagTipo.mina = false;
    this.flagTipo.municion = false;
    this.flagTipo.proyectil = false;
    this.flagTipo.cohete = false;
    this.flagTipo.misil = false;
    this.flagTipo.bomba = false;
    this.flagTipo.granada = false;

    if (this.artefacto.tipoArtefacto == "artefactos") {
      // console.log("soy un artefacto")
      this.marcarPropositoLibre();
    }
    if (this.artefacto.tipoArtefacto == "minas") {
      // console.log("soy una mina")
      this.flagTipo.mina = true;
      this.cargarOpcionesPropositoArtefacto(
        tiposEnvironment.propositoArtefacto.minas
      );
    }
    if (this.artefacto.tipoArtefacto == "municiones") {
      // console.log("soy una municion")
      this.flagTipo.municion = true;
      this.cargarOpcionesPropositoArtefacto(
        tiposEnvironment.propositoArtefacto.municiones
      );
    }
    if (this.artefacto.tipoArtefacto == "proyectiles") {
      // console.log("soy un proyectil")
      this.flagTipo.municion = true;
      this.flagTipo.proyectil = true;
      this.cargarOpcionesPropositoArtefacto(
        tiposEnvironment.propositoArtefacto.municiones
      );
    }
    if (this.artefacto.tipoArtefacto == "cohetes") {
      // console.log("soy un cobete")
      this.flagTipo.municion = true;
      this.flagTipo.cohete = true;
      this.cargarOpcionesPropositoArtefacto(
        tiposEnvironment.propositoArtefacto.municiones
      );
    }
    if (this.artefacto.tipoArtefacto == "misiles") {
      // console.log("soy un misil")
      this.flagTipo.municion = true;
      this.flagTipo.cohete = true;
      this.flagTipo.misil = true;
      this.cargarOpcionesPropositoArtefacto(
        tiposEnvironment.propositoArtefacto.municiones
      );
    }
    if (this.artefacto.tipoArtefacto == "bombasAviacion") {
      // console.log("soy una bomba")
      this.flagTipo.bomba = true;
      this.cargarOpcionesPropositoArtefacto(
        tiposEnvironment.propositoArtefacto.municiones
      );
    }
    if (this.artefacto.tipoArtefacto == "granadas") {
      // console.log("soy una granada")
      this.flagTipo.granada = true;
      this.cargarOpcionesPropositoArtefacto(
        tiposEnvironment.propositoArtefacto.municiones
      );
    }
    if (this.artefacto.tipoArtefacto == "espoletas") {
      // console.log("soy una espoleta")
      this.flagTipo.espoleta = true;
      this.cargarOpcionesPropositoArtefacto(
        tiposEnvironment.propositoArtefacto.espoletas
      );
    }
    if (this.artefacto.tipoArtefacto == "otros") {
      // console.log("soy un otro")
      this.marcarPropositoLibre;
    }
  }

  /**
   * Metodo que carga en el desplegable los propositos en funcion del tipo
   * @param propositos 
   * @see seleccionTipo() 
   */
  cargarOpcionesPropositoArtefacto(propositos: any[]) {
    let elemento = document.getElementById("propositoArtefacto");

    // while (elemento.firstChild) {
    //   elemento.firstChild.remove();
    // }
    propositos.forEach((e) => {
      let opcion = document.createElement("option");
      // console.log(opcion)
      opcion.value = e.key;
      opcion.innerText = e.value;

      elemento.appendChild(opcion);
      // console.log(opcion);
    });
  }

  /**
   * metod que muestra todos los tipos en caso que el tipo no sea de tipo especifico
   */
  marcarPropositoLibre() {
    this.cargarOpcionesPropositoArtefacto(
      tiposEnvironment.propositoArtefacto.municiones
        .concat(tiposEnvironment.propositoArtefacto.minas)
        .concat(tiposEnvironment.propositoArtefacto.espoletas)
    );
  }

  /**
   * Controla cuando cambia un campo del tipo forma y permite la visualizacion de las dimensiones en funcion de la forma
   *
   * cuando se produce un cambio tambien setea a cero los campos que no son propios de la forma
   */
  seleccionForma() {
    if (this.artefacto.forma != "" || !null) {
      this.flagForma.cilindro = false;
      this.flagForma.cubo = false;
      this.flagForma.esfera = false;
      tiposEnvironment.formas.forEach((forma) => {
        if (this.artefacto.forma == forma.key) {
          if (forma.forma == "cilindro") {
            this.artefacto.longitud = 0;
            this.artefacto.fondo = 0;
            this.flagForma.cilindro = true;
          } else if (forma.forma == "esfera") {
            this.artefacto.longitud = 0;
            this.artefacto.altura = 0;
            this.artefacto.fondo = 0;
            this.flagForma.esfera = true;
          } else {
            this.artefacto.diametroCalibre = 0;
            this.flagForma.cubo = true;
          }
        }
      });
    }
  }

  /**
   * recoge todos los inputs del formulario cuando hay un cambio y los almacena en un objeto para hacer despues un post, ademas lanza el control de los campos del formulario
   * @param payload objetos emitidos por cada subcomponente formulario cuando se produce un cambio
   */
  asignarArtefacto(payload: any) {
    this.artefacto = Object.assign({}, this.artefacto, payload);
    this.seleccionForma();
    this.seleccionTipo();
    // console.log("artefacto");
    // console.log(this.artefacto);

    // console.log(this.artefacto);
  }

  /**
   * salva un artefacto con el service ficha, almacena tambien las fotos y documentos en firebase con el servicio de los componentes hijos
   */
  salvarArtefacto() {
    this.id = undefined;
    if (!this.artefacto) {
      Swal.fire({
        title: "Error",
        text: "Debe introducir los campos",
        icon: "error",
      });
    }
    this.fichaService.postArtefacto(this.artefacto).subscribe((response) => {
      let url = response._links.self.href;
      for (let index = url.length - 1; index > 0; index--) {
        if (url[index] !== "/") {
          if (this.id == undefined) {
            this.id = url[index];
          } else {
            this.id = url[index] + this.id;
          }
        } else {
          break;
        }
      }
      this.hijo.subirImagenes(this.id);
      this.hijoArchivo.subirArchivos(this.id);
    });
  }
}
