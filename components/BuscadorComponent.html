import { Component, OnInit } from "@angular/core";
import { tiposEnvironment } from "src/environments/tiposEnvironment";
import { SearchCriteriaImpl } from "../models/search/search-criteriaImpl";
import { CriteriosBusquedaService } from "../services/criterios-busqueda.service";
import { faArrowDown, faArrowUp } from "@fortawesome/free-solid-svg-icons";

/**
 * componente que construye el objeto con los parametros de busqueda a partir de los inputs introducidos por el usuario
 *
 * Estos parametros se pasan al listado de fihcas para que presente solo las coincidencias obtenidas del Back
 */
@Component({
  selector: "app-buscador",
  templateUrl: "./buscador.component.html",
  styleUrls: ["./buscador.component.css"],
})
export class BuscadorComponent implements OnInit {
  searchCriteria: SearchCriteriaImpl = new SearchCriteriaImpl();
  searchCriteriaCache: SearchCriteriaImpl = this.searchCriteria;

  // criterios seleccionables en desplegables
  tipoArtefactos = tiposEnvironment.tipoArtefacto;

  formas = tiposEnvironment.formas;

  //iconos de Fawesome
  topMin = faArrowDown;
  topMax = faArrowUp;

  // criterios de dimensiones
  // para diametro
  diametroCalibreMax: number;
  diametroCalibreMin: number;
  // para longitud
  longitudMax: number;
  longitudMin: number;
  // para fondo
  fondoMax: number;
  fondoMin: number;
  // para altura
  alturaMax: number;
  alturaMin: number;
  //otros criterios de Artefacto
  colores = tiposEnvironment.colores;
  fabricacion = tiposEnvironment.fabricadoEn;
  propositoArtefacto = tiposEnvironment.propositoArtefacto.municiones
    .concat(tiposEnvironment.propositoArtefacto.espoletas)
    .concat(tiposEnvironment.propositoArtefacto.minas);

  //de minas
  materialEnvuelta = tiposEnvironment.materialEnvuelta;
  situacionEspoleta = tiposEnvironment.situacionEspoleta;

  // De municion
  longitudEspoletadoMax: number;
  longitudEspoletadoMin: number;
  //de proyectil
  tipoCulote = tiposEnvironment.tipoCulote;
  // de bombas de aviacion
  pinturaProteccionTermica: boolean;
  distanciaAnillasMax: number;
  distanciaAnillasMin: number;
  tipoEstabilizador: string;
  venterolEnEstabilizador: boolean;
  // de Cohete
  colorBandaMotor = tiposEnvironment.colores;
  // de Misil
  sistemaGuia = tiposEnvironment.sistemaGuia;
  // de Granada
  medioLanzamiento = tiposEnvironment.mediosLanzamiento;

  // de espoleta
  sistemaIniciacion = tiposEnvironment.sistemaIniciacion;

  //para la presentacion
  mostrarResultado: boolean = false;

  constructor(private criteriosService: CriteriosBusquedaService) {}

  ngOnInit() {}

  /**
   * metodo que desencadena la signacion de los parametros ultimos introducidos eliminando los borrados o deseleccionados
   */
  buscar() {
    this.searchCriteriaCache = new SearchCriteriaImpl();
    Object.keys(this.searchCriteria).map((key) => {
      if (this.searchCriteria[key] != "") {
        Object.defineProperty(this.searchCriteriaCache, key, {
          value: this.searchCriteria[key],
          writable: true,
          enumerable: true,
          configurable: true,
        });
      }
    });
    // console.log(JSON.stringify(this.searchCriteriaCache));
    this.mostrarResultado = true;
    this.criteriosService.onChangeCriteria(this.searchCriteriaCache);
  }
}
