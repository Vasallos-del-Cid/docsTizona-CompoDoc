import { THIS_EXPR } from "@angular/compiler/src/output/output_ast";
import { Component, OnInit } from "@angular/core";
import { ActivatedRoute, Router, RouterLinkActive } from "@angular/router";
import { Archivo } from "src/app/archivos/models/Archivo";
import { ArchivosService } from "src/app/archivos/service/archivos.service";
import { FichavipService } from "src/app/fichas-vip/service/fichas-vip.service";
import { Fotografia } from "src/app/fotografias/models/fotografia";
import { FotografiasService } from "src/app/fotografias/service/fotografias.service";
import { ArtefactoInventario } from "src/app/inventario/models/ArtefactoInventario";
import { Sala } from "src/app/inventario/models/sala";
import { InventarioService } from "src/app/inventario/service/inventario.service";
import { LoginService } from "src/app/seguridad/service/login.service";
import { FichaService } from "../services/ficha.service";
import Swal from "sweetalert2";
import { catchError } from "rxjs/operators";
import { throwError } from "rxjs";

/**
 * Componente de presentacion de una ficha completa
 */
@Component({
  selector: "app-ficha-completa",
  templateUrl: "./ficha-completa.component.html",
  styleUrls: ["./ficha-completa.component.css"],
})
export class FichaCompletaComponent implements OnInit {
  id: number = this.activatedRoute.snapshot.params["id"];
  artefacto: any;
  imagenPrincipal: Fotografia;
  imagenesSecundarias: Fotografia[] = [];
  imagenesCroquis: Fotografia[] = [];
  archivos: Archivo[] = [];
  atributos: string[][] = [];
  atributo: string[] = [];
  imagenModal: string;

  isLoggedUser;
  isLoggedAdmin;

  fichaVipFlag = false;

  artefactoInventario: ArtefactoInventario = new ArtefactoInventario();
  salasDisponibles: Sala[] = [];
  salaNueva: Sala;
  numeroInventario: string;

  constructor(
    private fotografiasService: FotografiasService,
    private fichaService: FichaService,
    private archivosService: ArchivosService,
    private fichasVipService: FichavipService,
    private activatedRoute: ActivatedRoute,
    private login: LoginService,
    private router: Router,
    private inventarioService: InventarioService
  ) {}

  ngOnInit() {
    this.fotografiasService.iniciarFirebase();
    this.getArtefacto();

    this.login.getIsLoggedFlagObs().subscribe((flag) => {
      this.isLoggedUser = flag;
    });
    this.login.getIsAdminFlagObs().subscribe((flag) => {
      this.isLoggedAdmin = flag;
    });

    this.fichaVipFlag = Boolean(
      this.fichasVipService.esVip(this.id).subscribe()
    );
    this.getSalasDisponibles();
  }

  /**
   * Obtener la URL de una imagen al hacer click
   * */
  getImagenURL(event): void {
    // console.log(event.target.src);
    this.imagenModal = event.target.src;
  }

  /**
   * Obtener los datos y fotografias del artefacto haciendo una llamada a la API y a firebase.
   */
  getArtefacto(): void {
    this.fichaService.getFicha(this.id).subscribe((response) => {
      let id = this.fichaService.getIdFicha(response);

      this.artefacto = this.fichaService.mapearFicha(response, id);
      this.pintarDatosArtefacto(this.artefacto);
    });

    // Obtener las imagenes del artefacto de Firebase.
    this.fotografiasService
      .getFotografiaPrincipal(this.id)
      .subscribe((response) => {
        this.imagenPrincipal = response;
      });
    this.fotografiasService
      .getFotografiasSecundarias(this.id)
      .subscribe((response) => {
        this.imagenesSecundarias = response;
      });
    this.fotografiasService
      .getFotografiasCroquis(this.id)
      .subscribe((response) => {
        this.imagenesCroquis = response;
      });
    this.archivosService.getArchivos(this.id).subscribe((response) => {
      // console.log(response);
      this.archivos = response;
    });
  }

  /**
   * metodo que transforma los campos del objeto recuperado en funcion de la respuesta del back
   * @param artefacto respuesta de la API
   */
  pintarDatosArtefacto(artefacto: any): void {
    // console.log(artefacto);
    //Recorrer los atributos del objeto como si fuesen pares clave-valor de un map.
    Object.keys(this.artefacto).map((key) => {
      // Recorrer el objeto artefacto cogiendo cada atributo y metiendolo en el array atributos.
      //Tambin cambio de los nombres de variable compuestos.
      if (key.toUpperCase() == "MARCASPINTURA") {
        this.atributo = ["MARCAS DE PINTURA", this.artefacto[key]];
      } else if (key.toUpperCase() == "MARCASFRIO") {
        this.atributo = ["MARCAS EN FRIO", this.artefacto[key]];
      } else if (key.toUpperCase() == "DIAMETROCALIBRE") {
        this.atributo = ["DIAMETRO / CALIBRE", this.artefacto[key]];
      } else if (key.toUpperCase() == "TIPOARTEFACTO") {
        this.atributo = ["TIPO DE ARTEFACTO", this.artefacto[key]];
      } else if (key.toUpperCase() == "PROPOSITOARTEFACTO") {
        this.atributo = ["PROPOSITO DEL ARTEFACTO", this.artefacto[key]];
      } else if (key.toUpperCase() == "MATERIALENVUELTA") {
        this.atributo = ["MATERIAL DE LA ENVUELTA", this.artefacto[key]];
      } else if (key.toUpperCase() == "SITUACIONESPOLETA") {
        this.atributo = ["SITUACIN DE LA ESPOLETA", this.artefacto[key]];
      } else if (key.toUpperCase() == "ID") {
        this.atributo = [null, null];
      } else if (key.toUpperCase() == "_LINKS") {
        this.atributo = [null, null];
      } else {
        this.atributo = [key.toUpperCase(), this.artefacto[key]];
      }
      if (this.artefacto[key] == 0) {
        this.atributo = [null, null];
      }

      this.atributos.push(this.atributo);
    });
  }

  /**
   * elimina una ficha y sus imagenes de back y servidor de imagenes
   */
  eliminarFicha(): void {
    Swal.fire({
      title: "Atencin",
      text: "Esta seguro de que quiere eliminar esta ficha?",
      icon: "info",
      showDenyButton: true,
      confirmButtonText: "Si, Eliminar",
      denyButtonText: "Cancelar",
    }).then((result) => {
      if (result.isConfirmed) {
        this.fichaService.deleteFicha(this.id).subscribe();
      }
    });
  }

  /**
   * cambia el estado de una ficha para que se muestre a los usuarios
   */
  validarFicha() {
    this.artefacto.validada = true;
    this.fichaService.updateFicha(this.artefacto, this.id).subscribe();
  }

  /**
   * recupera las salas para el inventario
   */
  getSalasDisponibles(): void {
    this.inventarioService.getSalas().subscribe((response) => {
      this.salasDisponibles = this.inventarioService.mapearSalas(response);
      // console.log("salas disponibles", this.salasDisponibles);
    });
  }

  /**
   * permite agregar una ficha en el inventario
   */
  agregarAInventario(): void {
    // console.log("en agregar a inventario");
    this.artefactoInventario.artefactoImpl = this.artefacto._links.self.href;
    this.inventarioService
      .postArtefactoInventario(this.artefactoInventario)
      .subscribe();
  }
}
