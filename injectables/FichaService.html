import { HttpClient, HttpHeaders } from "@angular/common/http";
import { Injectable, Type } from "@angular/core";
import { Observable, throwError } from "rxjs";
import { map, catchError } from "rxjs/operators";
import { CriptoService } from "src/app/seguridad/service/cripto.service";
import { environment } from "src/environments/environment";
import { Artefacto } from "../models/artefactos/Artefacto";
import { ArtefactoImpl } from "../models/artefactos/ArtefactoImpl";
import { SearchCriteria } from "../models/search/search-criteria";
import Swal from "sweetalert2";

const cabecera = {
  headers: new HttpHeaders({ "Content-Type": "application/json" }),
};

/**
 * servicio de comunicacion http con back para el CRUD de fichas
 */
@Injectable({
  providedIn: "root",
})
export class FichaService {
  private urlEndPoint: string = environment.urlAPI + `/artefactos`;
  private urlPostEndpoint: string = environment.urlAPI;

  constructor(private http: HttpClient, private cripto: CriptoService) {}

  /**
   * post
   */
  postArtefacto(artefacto: Artefacto): Observable<any> {
    artefacto = this.cripto.encriptarFicha(artefacto);
    return this.http
      .post(
        `${this.urlPostEndpoint}/${artefacto.tipoArtefacto}`,
        artefacto,
        cabecera
      )
      .pipe(
        map((response) => {
          if (response) {
            console.log("ok: ", response);
            Swal.fire({
              title: "Ficha Almacenada",
              icon: "success",
            });
          }
        }),
        catchError((e) => {
          Swal.fire({
            title: "Error",
            text: "¡Vaya! No se ha podido almacenar la ficha. Si el problema persiste contacte al administrador",
            icon: "error",
          });
          console.log("error", e);
          if (e.status === 400) {
            return throwError(e);
          }
          if (e.error.mensaje) {
            console.error(e.error.mensaje);
          }
          return throwError(e);
        })
      );
  }

  /**
   * boora fichas por id
   * @param id number id de ficha
   * @returns respuesta http
   */
  deleteFicha(id: number): Observable<any> {
    return this.http.delete(this.urlEndPoint + "/" + id, cabecera).pipe(
      map((response) => {
        console.log("Eliminando: ok");
        Swal.fire({
          title: "Ficha Eliminada",
          icon: "success",
        });
      }),
      catchError((e) => {
        Swal.fire({
          title: "Error",
          text: "¡Vaya! No se ha podido eliminar la ficha. Si el problema persiste contacte al administrador",
          icon: "error",
        });
        console.log("error", e);
        if (e.status === 400) {
          return throwError(e);
        }
        if (e.error.mensaje) {
          console.error(e.error.mensaje);
        }
        return throwError(e);
      })
    );
  }

  /**
   * actualiza una ficha
   * @param artefactoModif
   * @param id
   * @returns
   */
  updateFicha(artefactoModif: Artefacto, id: number): Observable<any> {
    // console.log("ficha a validar")
    // console.log(artefactoModif)
    // console.log(`${this.urlPostEndpoint}/${artefactoModif.tipoArtefacto}/${id}`)
    return this.http
      .patch(
        `${this.urlPostEndpoint}/${artefactoModif.tipoArtefacto}/${id}`,
        artefactoModif,
        cabecera
      )
      .pipe(
        map((response) => {
          if (response) {
            console.log("ok: ", response);
            Swal.fire({
              title: "Ficha Almacenada",
              text: "A patir de ahora la ficha estará disponible para su busqueda por los usuarios",
              icon: "success",
            });
          }
        }),
        catchError((e) => {
          Swal.fire({
            title: "Error",
            text: "¡Vaya! No se ha podido almacenar la ficha. Si el problema persiste contacte al administrador",
            icon: "error",
          });
          console.log("error", e);
          if (e.status === 400) {
            return throwError(e);
          }
          if (e.error.mensaje) {
            console.error(e.error.mensaje);
          }
          return throwError(e);
        })
      );
  }

  /**
   * recupera todas las fichas
   * @returns respuesta json del back
   */
  getFichas(): Observable<any> {
    // console.log(`Url acceso a la peticion ${this.urlEndPoint}`);
    let objetoApiFichas = this.http.get<any>(this.urlEndPoint, cabecera);

    // console.log(objetoApiFichas);
    return objetoApiFichas;
  }

  /**
   * resupera solo las fichas validadas para mostrar al usuario comun
   * @returns
   */
  getFichasSinValidar() {
    let objetoApiFichas = this.http.get<any>(
      this.urlEndPoint + "/search/findAllByValidada?validada=false",
      cabecera
    );
    console.log(objetoApiFichas);
    return objetoApiFichas;
  }

  /**
   * construyye fichas presentables a partir de la respuesta cobtenida
   * @param respuestaApi
   * @returns
   */
  extraerFichas(respuestaApi: any): Artefacto[] {
    const fichas: Artefacto[] = [];

    this.recorrerElementos(respuestaApi._embedded.artefactos, fichas);
    this.recorrerElementos(respuestaApi._embedded.minas, fichas);
    this.recorrerElementos(respuestaApi._embedded.municiones, fichas);
    this.recorrerElementos(respuestaApi._embedded.proyectiles, fichas);
    this.recorrerElementos(respuestaApi._embedded.bombasAviacion, fichas);
    this.recorrerElementos(respuestaApi._embedded.cohetes, fichas);
    this.recorrerElementos(respuestaApi._embedded.misiles, fichas);
    this.recorrerElementos(respuestaApi._embedded.granadas, fichas);
    this.recorrerElementos(respuestaApi._embedded.espoletas, fichas);
    // console.log("Longitud del array")
    // console.log(fichas.length)
    return fichas;
  }

  /**
   * metodo que recorre los objetos individuales de los array de la respuesta de la llamada a getFichas
   * @param arrayRespuesta
   * @param fichas
   */
  recorrerElementos(arrayRespuesta, fichas) {
    if (arrayRespuesta) {
      arrayRespuesta.forEach((p) => {
        let id = this.getIdFicha(p);
        fichas.push(this.mapearFicha(p, id));
      });
    }
  }

  /**
   * metodo qeu transforma un objeto json de la api a una ficha front
   * @param fichaApi
   * @param id
   * @returns
   */
  mapearFicha(fichaApi: any, id: String): Artefacto {
    let objeto = Object.create(ArtefactoImpl);
    Object.assign<Artefacto, Artefacto>(objeto, fichaApi);
    Object.defineProperty(objeto, "id", {
      value: null,
      writable: true,
      enumerable: true,
      configurable: true,
    });
    objeto.id = id;
    if (objeto.validada == false) {
      objeto = this.cripto.desencriptarFicha(objeto);
    }
    return objeto;
  }

  /**
   * metodo qeue recupera el id a partir de una url de la respuesta del back
   * @param p
   * @returns
   */
  getIdFicha(p: any): String {
    let url = p._links.self.href;
    let trozos = url.split("/");
    // console.log(`trozos: ${ trozos }`);
    return trozos[trozos.length - 1];
  }

  /**
   * metodo que recupera las fichas de la llamada al metodo personalizado con los parametros de busqueda(filtros)
   * @param searchCriteria
   * @returns
   */
  getFichasFiltradas(searchCriteria: SearchCriteria): Observable<any> {
    let objetoApiFichas = this.http.get<any>(
      this.urlEndPoint +
        "/search/filtrar?" +
        this.construirParametros(searchCriteria),
      cabecera
    );
    return objetoApiFichas;
  }

  /**
   * metodo que construye el fragemento de la url con los parametros selecionados a partir de un objeto con criterios key:value
   * @param searchCriteria
   * @returns string "?param=valor..."
   */
  construirParametros(searchCriteria: SearchCriteria): string {
    let parametros = "";
    Object.keys(searchCriteria).map((key) => {
      parametros += key + "=" + searchCriteria[key] + "&";
      // console.log(parametros);
    });
    return parametros.slice(0, parametros.length - 1);
  }

  /**
   * recupera una ficha por el id
   * @param id
   * @returns
   */
  getFicha(id: number): Observable<any> {
    return this.http.get(this.urlEndPoint + "/" + id, cabecera);
  }

  /**
   * comprueba si una ficha esta validada o no por su id
   * @param id
   * @returns
   */
  estaValidada(id: string): Observable<any> {
    return this.http.get(
      this.urlEndPoint + `/search/validada?validada=true&id=${id}`
    );
  }
}
